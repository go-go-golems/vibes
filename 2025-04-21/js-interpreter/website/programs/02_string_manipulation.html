Program Code

This program tests various string manipulation methods in JavaScript: length, case conversion, substring, replace, split, and trim.

// String manipulation
const str = "Hello, World!";

const length = str.length;
const uppercase = str.toUpperCase();
const lowercase = str.toLowerCase();
const substring = str.substring(0, 5);
const replaced = str.replace("World", "JavaScript");
const split = str.split(", ");
const trimmed = "  trim me  ".trim();

console.log("Original:", str);
console.log("Length:", length);
console.log("Uppercase:", uppercase);
console.log("Lowercase:", lowercase);
console.log("Substring:", substring);
console.log("Replaced:", replaced);
console.log("Split:", split);
console.log("Trimmed:", trimmed);
