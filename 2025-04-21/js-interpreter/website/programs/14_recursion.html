Program Code

This program tests recursive functions in JavaScript, including factorial calculation, Fibonacci sequence, and tree traversal.

// Recursion
// Factorial calculation
function factorial(n) {
    if (n <= 1) return 1;
    return n * factorial(n - 1);
}

// Fibonacci sequence
function fibonacci(n) {
    if (n <= 1) return n;
    return fibonacci(n - 1) + fibonacci(n - 2);
}

// Tree traversal with recursion
const tree = {
    value: 1,
    left: {
        value: 2,
        left: { value: 4, left: null, right: null },
        right: { value: 5, left: null, right: null }
    },
    right: {
        value: 3,
        left: { value: 6, left: null, right: null },
        right: { value: 7, left: null, right: null }
    }
};

// In-order traversal (left, root, right)
function inOrderTraversal(node, result = []) {
    if (node === null) return result;
    
    inOrderTraversal(node.left, result);
    result.push(node.value);
    inOrderTraversal(node.right, result);
    
    return result;
}

// Sum of array using recursion
function sumArray(arr, index = 0) {
    if (index >= arr.length) return 0;
    return arr[index] + sumArray(arr, index + 1);
}

console.log("Factorial of 5:", factorial(5));
console.log("Fibonacci of 7:", fibonacci(7));
console.log("In-order traversal:", inOrderTraversal(tree));
console.log("Sum of array [1,2,3,4,5]:", sumArray([1, 2, 3, 4, 5]));
