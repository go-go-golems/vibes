Program Code

This program tests Set objects in JavaScript, including creation, manipulation, and set operations.

// Sets
// Creating a Set
const fruits = new Set(['apple', 'banana', 'orange', 'apple']);
console.log("Fruits set size:", fruits.size);
console.log("Fruits set has apple:", fruits.has('apple'));
console.log("Fruits set has grape:", fruits.has('grape'));

// Adding and deleting elements
fruits.add('grape');
fruits.add('mango');
fruits.delete('banana');
console.log("Updated fruits set size:", fruits.size);

// Iterating over a Set
console.log("Fruits set elements:");
fruits.forEach(fruit => {
    console.log("- " + fruit);
});

// Converting Set to Array
const fruitsArray = Array.from(fruits);
console.log("Fruits array:", fruitsArray);

// Set operations (using arrays and Sets)
const setA = new Set([1, 2, 3, 4, 5]);
const setB = new Set([3, 4, 5, 6, 7]);

// Union
const union = new Set([...setA, ...setB]);
console.log("Union:", Array.from(union));

// Intersection
const intersection = new Set(
    [...setA].filter(x => setB.has(x))
);
console.log("Intersection:", Array.from(intersection));

// Difference (A - B)
const difference = new Set(
    [...setA].filter(x => !setB.has(x))
);
console.log("Difference (A - B):", Array.from(difference));

// Symmetric difference
const symmetricDifference = new Set(
    [...setA].filter(x => !setB.has(x)).concat([...setB].filter(x => !setA.has(x)))
);
console.log("Symmetric difference:", Array.from(symmetricDifference));
