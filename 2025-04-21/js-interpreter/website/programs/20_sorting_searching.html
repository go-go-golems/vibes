Program Code

This program tests sorting and searching algorithms in JavaScript, including array sorting, custom comparators, and binary search.

// Sorting and searching
// Basic array sorting
const numbers = [5, 2, 8, 1, 4];
const sortedNumbers = [...numbers].sort((a, b) => a - b);
console.log("Original numbers:", numbers);
console.log("Sorted numbers:", sortedNumbers);

// Sorting strings
const fruits = ["banana", "apple", "orange", "grape"];
const sortedFruits = [...fruits].sort();
console.log("Sorted fruits:", sortedFruits);

// Sorting objects
const people = [
    { name: "John", age: 30 },
    { name: "Alice", age: 25 },
    { name: "Bob", age: 35 }
];

// Sort by age
const sortedByAge = [...people].sort((a, b) => a.age - b.age);
console.log("Sorted by age:");
sortedByAge.forEach(person => {
    console.log(`${person.name}: ${person.age}`);
});

// Sort by name
const sortedByName = [...people].sort((a, b) => a.name.localeCompare(b.name));
console.log("Sorted by name:");
sortedByName.forEach(person => {
    console.log(`${person.name}: ${person.age}`);
});

// Binary search (on sorted array)
function binarySearch(arr, target) {
    let left = 0;
    let right = arr.length - 1;
    
    while (left <= right) {
        const mid = Math.floor((left + right) / 2);
        
        if (arr[mid] === target) {
            return mid;
        }
        
        if (arr[mid] < target) {
            left = mid + 1;
        } else {
            right = mid - 1;
        }
    }
    
    return -1;
}

const sortedArray = [1, 2, 4, 5, 8];
console.log("Binary search for 4:", binarySearch(sortedArray, 4));
console.log("Binary search for 3:", binarySearch(sortedArray, 3));

// Custom sorting with multiple criteria
const students = [
    { name: "Alice", grade: "A", age: 22 },
    { name: "Bob", grade: "B", age: 20 },
    { name: "Charlie", grade: "A", age: 21 },
    { name: "David", grade: "C", age: 20 }
];

// Sort by grade, then by age
const sortedStudents = [...students].sort((a, b) => {
    // First by grade
    if (a.grade !== b.grade) {
        return a.grade.localeCompare(b.grade);
    }
    // Then by age
    return a.age - b.age;
});

console.log("Students sorted by grade, then age:");
sortedStudents.forEach(student => {
    console.log(`${student.name}: Grade ${student.grade}, Age ${student.age}`);
});
