Program Code

This program tests destructuring assignment in JavaScript, including array destructuring, object destructuring, and nested destructuring.

// Destructuring
// Array destructuring
const numbers = [1, 2, 3, 4, 5];

// Basic array destructuring
const [first, second, ...rest] = numbers;
console.log("First:", first);
console.log("Second:", second);
console.log("Rest:", rest);

// Skipping elements
const [a, , c] = numbers;
console.log("a:", a);
console.log("c:", c);

// Default values
const [x = 10, y = 20, z = 30] = [1, 2];
console.log("x, y, z:", x, y, z);

// Swapping variables
let m = 1;
let n = 2;
[m, n] = [n, m];
console.log("Swapped m, n:", m, n);

// Object destructuring
const person = {
    name: "John",
    age: 30,
    city: "New York",
    country: "USA"
};

// Basic object destructuring
const { name, age } = person;
console.log("Name:", name);
console.log("Age:", age);

// Renaming properties
const { city: location, country: nation } = person;
console.log("Location:", location);
console.log("Nation:", nation);

// Default values
const { hobby = "Reading" } = person;
console.log("Hobby:", hobby);

// Nested destructuring
const user = {
    id: 1,
    details: {
        firstName: "Alice",
        lastName: "Smith",
        address: {
            street: "123 Main St",
            city: "Boston"
        }
    },
    hobbies: ["Reading", "Hiking", "Coding"]
};

// Extracting nested properties
const { 
    details: { 
        firstName, 
        lastName,
        address: { city: userCity }
    },
    hobbies: [firstHobby, ...otherHobbies]
} = user;

console.log("First name:", firstName);
console.log("Last name:", lastName);
console.log("User city:", userCity);
console.log("First hobby:", firstHobby);
console.log("Other hobbies:", otherHobbies);

// Function parameter destructuring
function printUserInfo({ name, age, city = "Unknown" }) {
    console.log(`${name}, ${age}, ${city}`);
}

printUserInfo(person);
