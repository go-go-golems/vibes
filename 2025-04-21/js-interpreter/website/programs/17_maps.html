Program Code

This program tests Map objects in JavaScript, including creation, manipulation, and various Map methods.

// Maps
// Creating a Map
const userMap = new Map();

// Adding key-value pairs
userMap.set('user1', { name: 'John', age: 25 });
userMap.set('user2', { name: 'Alice', age: 30 });
userMap.set('user3', { name: 'Bob', age: 35 });

// Getting values
const user1 = userMap.get('user1');
console.log("User1:", user1);

// Checking if a key exists
console.log("Has user2:", userMap.has('user2'));
console.log("Has user4:", userMap.has('user4'));

// Map size
console.log("Map size:", userMap.size);

// Deleting entries
userMap.delete('user2');
console.log("Map size after delete:", userMap.size);

// Iterating over a Map
console.log("Map entries:");
userMap.forEach((value, key) => {
    console.log(`${key}: ${value.name}, ${value.age}`);
});

// Creating a Map from an array of key-value pairs
const fruitPrices = new Map([
    ['apple', 1.2],
    ['banana', 0.9],
    ['orange', 1.5]
]);
console.log("Fruit prices map size:", fruitPrices.size);

// Getting all keys, values, and entries
const keys = Array.from(fruitPrices.keys());
const values = Array.from(fruitPrices.values());
const entries = Array.from(fruitPrices.entries());

console.log("Fruit keys:", keys);
console.log("Fruit values:", values);
console.log("Fruit entries:", entries);

// Using objects as keys (not possible with regular objects)
const obj1 = { id: 1 };
const obj2 = { id: 2 };

const objectMap = new Map();
objectMap.set(obj1, 'Object 1 value');
objectMap.set(obj2, 'Object 2 value');

console.log("Object map size:", objectMap.size);
console.log("Object 1 value:", objectMap.get(obj1));
